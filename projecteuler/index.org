  (defun eu-gen ()
    (interactive)
    (let* ((files (directory-files default-directory
				   nil
				   "^[0-9]+.org"))
	   (names (mapcar (lambda (x) (concat "./" x))
			  files))
	   (numbers (mapcar (lambda (x) (file-name-base x)) files)))
      (cl-loop for a in names
	       for i in numbers
	       do (insert (format "- [[%s][problem %s]]\n" a i)))))
